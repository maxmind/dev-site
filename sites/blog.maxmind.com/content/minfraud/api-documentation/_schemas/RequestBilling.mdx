import requestJson from '../_examples/request';

<MinFraudSchema
  json={requestJson.billing}
  jsonPointer="/billing"
  name="Request | Billing"
  services="*"
>
  This object contains the billing address and contact information provided by
  the end-user who initiated the event.

  <Property
    name="first_name"
    tags={{
      'Max Length': 255,
    }}
  >
    The first name of the end user as provided in their billing information.
  </Property>

  <Property
    name="last_name"
    tags={{
      'Max Length': 255,
    }}
  >
    The last name of the end user as provided in their billing information.
  </Property>

  <Property
    name="company"
    tags={{
      'Max Length': 255,
    }}
  >
    The company of the end user as provided in their billing information.
  </Property>

  <Property
    name="address"
    tags={{
      'Max Length': 255,
    }}
  >
    The first line of the user's billing address.
  </Property>

  <Property
    name="address_2"
    tags={{
      'Max Length': 255,
    }}
  >
    The second line of the user's billing address.
  </Property>

  <Property
    name="city"
    tags={{
      'Max Length': 255,
    }}
  >
    The city of the user's billing address.
  </Property>

  <Property
    name="region"
    tags={{
      'Max Length': 4,
    }}
  >
    The [ISO 3166-2 subdivision code](https://en.wikipedia.org/wiki/ISO_3166-2)
    for the user's billing address.
  </Property>

  <Property
    name="country"
    tags={{
      'Max Length': 2,
    }}
  >
    The two character [ISO 3166-1 alpha-2 country code](https://en.wikipedia.org/wiki/ISO_3166-1_alpha-2)
    of the user's billing address.
  </Property>

  <Property
    name="postal"
    tags={{
      'Max Length': 255,
    }}
  >
    The postal code of the user's billing address.
  </Property>

  <Property
    name="phone_number"
    tags={{
      'Max Length': 255,
    }}
  >
    The phone number without the country code for the user's billing address.
    Punctuation characters will be stripped. After stripping punctuation
    characters, the number must contain only digits.
  </Property>

  <Property
    name="phone_country_code"
    tags={{
      'Max Length': 4,
    }}
  >
    The country code for phone number associated with the user's billing
    address. If you provide this information then you must provide at least one
    digit.
  </Property>
</MinFraudSchema>
